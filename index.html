Use this demand function to determine the optimal price for max revenue, corresponding demand, and the max revenue possible given that demand

In[12]:
# Demand function: Demand = 324.90 - 107.33 * Price

# Define revenue function
def revenue(price):
    return price * (intercept - (slope * price))

# Find the optimal price for maximum revenue
# Since this is a quadratic function, the max occurs at the vertex
a = slope
b = intercept
optimal_price = -b / (2 * a)

# Calculate max revenue and corresponding demand
max_revenue = revenue(optimal_price)
optimal_demand = intercept - (slope * optimal_price)

print(f"Optimal Price for Max Revenue: ${optimal_price:.2f}")
print(f"Corresponding Demand: {optimal_demand:.0f} units")
print(f"Maximum Revenue: ${max_revenue:,.2f}")
